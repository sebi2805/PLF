BUDGET THIS YEAR = 
CALCULATE(
    SUM('MONTHLY PRODUCTS FINANCIALS'[AMOUNT]), 
    YEAR('MONTHLY PRODUCTS FINANCIALS'[MONTH]) = YEAR(TODAY()),
    'MONTHLY PRODUCTS FINANCIALS'[FINANCIAL CATEGORY] = "BUDGET"
)


Budget - Revenue Change % = 
IF(
    [BUDGET THIS YEAR] = 0, 
    BLANK(),
    IF(
        [Revenue This Year] = 0, 
        BLANK(),
        ([Revenue This Year] - [BUDGET THIS YEAR]) / [BUDGET THIS YEAR]
    )
)


REVENUE LAST YEAR = 
CALCULATE(
    SUM('MONTHLY PRODUCTS FINANCIALS'[AMOUNT]), 
    YEAR('MONTHLY PRODUCTS FINANCIALS'[MONTH]) = YEAR(TODAY()) - 1,
    'MONTHLY PRODUCTS FINANCIALS'[FINANCIAL CATEGORY] = "REVENUE" 
)


REVENUE THIS YEAR = 
CALCULATE(
    SUM('MONTHLY PRODUCTS FINANCIALS'[AMOUNT]), 
    YEAR('MONTHLY PRODUCTS FINANCIALS'[MONTH]) = YEAR(TODAY()),
    ('MONTHLY PRODUCTS FINANCIALS'[FINANCIAL CATEGORY] = "REVENUE")||('MONTHLY PRODUCTS FINANCIALS'[FINANCIAL CATEGORY] = "COS")
)
